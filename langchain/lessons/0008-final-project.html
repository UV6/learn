3.1 State 定义

# src/state.py
from typing import TypedDict, Annotated, Literal
from langgraph.graph.message import add_messages

class CSAgentState(TypedDict):
    messages: Annotated[list, add_messages]
    user_query: str
    intent: str                    # "faq" | "order" | "general"
    faq_result: str                # FAQ Agent 的检索结果
    order_result: str              # Order Agent 的处理结果
    pending_approval: dict | None  # 待审批的操作
    final_answer: str
    next: str                      # supervisor 的路由决策

3.2 FAQ Agent 实现

# src/agents/faq_agent.py
from langchain_community.document_loaders import TextLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_community.embeddings import HuggingFaceEmbeddings
from langchain_community.vectorstores import FAISS
from langchain.chains import create_retrieval_chain
from langchain.chains.combine_documents import create_stuff_documents_chain
from langchain_core.prompts import ChatPromptTemplate

def build_faq_chain(model):
    # 加载产品文Knowledge
    loader = TextLoader("knowledge/products.txt", encoding="utf-8")
    docs = loader.load()

    splitter = RecursiveCharacterTextSplitter(chunk_size=300, chunk_overlap=50)
    chunks = splitter.split_documents(docs)

    embeddings = HuggingFaceEmbeddings(model_name="BAAI/bge-small-zh-v1.5")
    vectorstore = FAISS.from_documents(chunks, embeddings)
    retriever = vectorstore.as_retriever(search_kwargs={"k": 3})

    prompt = ChatPromptTemplate.from_messages([
        ("system", """基于以下产品信息回答问题。找不到就说不知道。
产品信息：{context}"""),
        ("human", "{input}"),
    ])

    combine_chain = create_stuff_documents_chain(model, prompt)
    return create_retrieval_chain(retriever, combine_chain)

def faq_agent_node(state):
    chain = build_faq_chain(model)
    result = chain.invoke({"input": state["user_query"]})
    return {"faq_result": result["answer"]}

3.3 Order Agent 实现

# src/agents/order_agent.py
from langchain_core.tools import BaseTool
from langgraph.types import interrupt
from pydantic import BaseModel, Field

# --- Tools ---
class OrderLookupInput(BaseModel):
    order_id: str = Field(description="订单号，格式 ORD-XXXXX")

class OrderLookupTool(BaseTool):
    name = "lookup_order"
    description = "查询订单状态。输入订单号。"
    args_schema = OrderLookupInput

    def _run(self, order_id: str) -> str:
        # 模拟数据库查询
        orders = {
            "ORD-12345": "已发货，预计6月30日到达",
            "ORD-67890": "处理中，预计7月2日发货",
        }
        return orders.get(order_id, f"未找到订单 {order_id}")

class RefundInput(BaseModel):
    order_id: str = Field(description="要退款的订单号")
    reason: str = Field(description="退款原因")

class RefundTool(BaseTool):
    name = "process_refund"
    description = "处理退款。需要主管审批。"
    args_schema = RefundInput

    def _run(self, order_id: str, reason: str) -> str:
        # 触发人工审批
        approval = interrupt(
            f"退款请求：订单 {order_id}，原因：{reason}。批准？"
        )
        if approval == "approved":
            return f"订单 {order_id} 已退款。退款金额将在3-5个工作日到账。"
        return f"订单 {order_id} 退款已拒绝。"

def order_agent_node(state):
    # 用 create_react_agent + order tools
    agent = create_react_agent(
        model, [OrderLookupTool(), RefundTool()],
        system_prompt="你是客服订单专员。处理查询和退款。"
    )
    result = agent.invoke({"messages": state["messages"]})
    return {"order_result": result["messages"][-1].content}

3.4 Supervisor 实现

# src/supervisor.py
from langgraph.graph import StateGraph, START, END

MEMBERS = ["faq_agent", "order_agent", "general_agent"]

SUPERVISOR_PROMPT = """你是客服路由系统。根据用户问题，决定分发给谁：
- faq_agent: 产品问题、功能咨询、价格、使用说明
- order_agent: 订单查询、退款、物流
- general_agent: 问候、闲聊、无法分类的问题

回复格式：只回复 agent 名称或 FINISH。"""

def supervisor_node(state):
    response = model.invoke([
        {"role": "system", "content": SUPERVISOR_PROMPT},
        {"role": "user", "content": state["user_query"]},
    ])
    return {"next": response.content.strip(), "intent": response.content.strip()}

def router(state):
    if state["next"] == "FINISH":
        return "__end__"
    if state["next"] in MEMBERS:
        return state["next"]
    return "general_agent"  # fallback

# --- 构建图 ---
from src.agents.faq_agent import faq_agent_node
from src.agents.order_agent import order_agent_node
from src.agents.general_agent import general_agent_node
from langgraph.checkpoint.memory import MemorySaver

builder = StateGraph(CSAgentState)
builder.add_node("supervisor", supervisor_node)
builder.add_node("faq_agent", faq_agent_node)
builder.add_node("order_agent", order_agent_node)
builder.add_node("general_agent", general_agent_node)

builder.add_edge(START, "supervisor")
builder.add_conditional_edges("supervisor", router)
builder.add_edge("faq_agent", "supervisor")     # 干完活回 supervisor
builder.add_edge("order_agent", "supervisor")
builder.add_edge("general_agent", "supervisor")

graph = builder.compile(checkpointer=MemorySaver())

3.5 交互式入口

# src/main.py
if __name__ == "__main__":
    config = {"configurable": {"thread_id": "demo-001"}}

    print("🤖 智能客服系统已启动 (输入 'quit' 退出)\n")

    while True:
        query = input("👤 你: ")
        if query.lower() == "quit":
            break

        print("🤖 客服: ", end="", flush=True)

        for event in graph.stream(
            {"messages": [{"role": "user", "content": query}],
             "user_query": query},
            config,
            stream_mode="updates",
        ):
            # 实时展示 agent 切换
            node = list(event.keys())[0]
            if node == "supervisor":
                intent = event[node].get("intent", "")
                print(f"\n[路由到: {intent}]", end="")
            elif node in MEMBERS:
                print(f"\n[{node}] ", end="")

        print("\n")

4. README.md 模板（面试用）

# 智能客服系统 (Intelligent CS Agent)

基于 LangChain + LangGraph 的多 Agent 协作客服系统，支持产品FAQ、订单处理、退款审批。

## 架构
- **Supervisor Agent**: 意图识别 + 路由分发
- **FAQ Agent**: RAG 检索增强（FAISS + BGE Embedding）
- **Order Agent**: 订单查询 + 退款（Human-in-the-Loop 审批）
- **General Agent**: 通用对话兜底

## 技术栈
- LangChain: Agent 创建、Tool 定义、RAG 管道
- LangGraph: StateGraph、条件路由、Checkpointing
- FAISS + HuggingFace Embeddings: 本地向量检索
- DeepSeek API: LLM 推理

## 核心特性
- ✅ 多 Agent 协作（Supervisor 路由模式）
- ✅ Human-in-the-Loop（退款审批）
- ✅ Streaming 响应（实时反馈）
- ✅ 持久化对话记忆（MemorySaver）
- ✅ 可观测性（Callback + token tracking）
